#include <iostream>
#include <math.h>
#include <cmath>
using namespace std;

int main(void) {

    const int size = 12;
    int m[size] = { 0,52,6,29,3,-17,0,31,14,-1,19,4 };
    bool nul = false;

    for (int i = 0; i < 12; i++)
    {
        if (m[i] == 0)
        {
            cout << "zero is at number: " << i + 1 << endl;
            nul = true;
        }
    }

    if (!nul)
    {
        cout << "has no zeros" << endl;
    }
}
