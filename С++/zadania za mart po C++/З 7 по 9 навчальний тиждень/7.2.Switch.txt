#include <iostream>

using namespace std;

int monthLength(int year, int month) {
    int daysInMonth;
    switch (month) {
    case 2:
        if (year % 4 == 0 && (year % 100 != 0 || year % 400 == 0)) {
            daysInMonth = 29;
        }
        else {
            daysInMonth = 28;
        }
        break;
    case 4:
    case 6:
    case 9:
    case 11:
        daysInMonth = 30;
        break;
    default:
        daysInMonth = 31;
        break;
    }
    return daysInMonth;
}

int main(void) {
    for (int yr = 2000; yr < 2002; yr++) {
        for (int mo = 1; mo <= 12; mo++) {
            cout << monthLength(yr, mo) << " ";
        }
        cout << endl;
    }
    return 0;
}
