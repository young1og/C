#include <iostream>

using namespace std;

int main() {
    const int size = 5;
    float* arr = new float[size];

    // заполнение массива случайными значениями
    for (int i = 0; i < size; i++) {
        arr[i] = static_cast<float>(rand()) / RAND_MAX * 4 - 2;
    }

    // подсчет количества положительных и отрицательных элементов
    int positive_count = 0, negative_count = 0;
    for (int i = 0; i < size; i++) {
        if (arr[i] > 0) {
            positive_count++;
        }
        else if (arr[i] < 0) {
            negative_count++;
        }
    }

    // вывод результатов
    cout << "Array elements:" << std::endl;
    for (int i = 0; i < size; i++) {
        cout << arr[i] << " ";
    }
    cout << std::endl;
    cout << "Number of positive : " << positive_count << endl;
    cout << "Number of negative : " << negative_count << endl;

    delete[] arr;
    return 0;
}
